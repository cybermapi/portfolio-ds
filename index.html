Função conversor


class Relogio {
  //classe
  double _h1 = 0; //variáveis
  double? _r;

  double get h1 => _h1; //getter
  set h1(double novoh1) => _h1 = novoh1; //setter

  double? get r => _r;

  void minutos() {
    //método de conversão de horas em minutos
    this._r = this._h1 * 60;
  }

  void segundos() {
    //método de conversão de horas em segundos
    this._r = this._h1 * 3600;
  }
}

void main() {
  var relogio = Relogio();

  relogio.h1 = 4;

  relogio.minutos();
  print("${relogio.h1} horas equivale a ${relogio.r} minutos");

  relogio.segundos();
  print("${relogio.h1} horas equivale a ${relogio.r} segundos");
}

Metodologia:

Este código implementa uma classe chamada Relogio, que realiza conversões de tempo, transformando horas em minutos ou segundos. A classe encapsula os dados utilizando variáveis privadas (_h1 e _r) e acessa essas informações através de getters e setters. Os métodos minutos() e segundos() calculam os equivalentes de horas em minutos e segundos, respectivamente, armazenando os resultados na variável _r. No método main, um objeto da classe é instanciado, o valor de horas é definido e os métodos são chamados para realizar as conversões, exibindo os resultados no console.

Classe abstrata


abstract class Funcionario {
  void salario();
}

class Gerente extends Funcionario {
  late double horas, valorHora;

  void salario() {
    var aPagar = horas * valorHora;
    print('O valor a ser pago ao gerente é de R\$: $aPagar,00');
  }
}

class Assistente extends Funcionario {
  late double horas, valorHora;

  void salario() {
    var aPagar = horas * valorHora;
    print('O valor a ser pago ao assistente é de R\$: $aPagar,00');
  }
}

void main() {
  var colaborador1 = Gerente();
  colaborador1.horas = 220;
  colaborador1.valorHora = 20;
  colaborador1.salario();

  var colaborador2 = Assistente();
  colaborador2.horas = 220;
  colaborador2.valorHora = 10;
  colaborador2.salario();
}

Metodologia:

Este código apresenta uma abordagem de herança em programação orientada a objetos. A classe abstrata Funcionario define o método salario() como uma interface que deve ser implementada pelas classes derivadas. As classes concretas Gerente e Assistente estendem a classe abstrata, implementando o método salario() para calcular e exibir o pagamento correspondente ao número de horas trabalhadas multiplicado pelo valor por hora. No método main, são criadas instâncias de Gerente e Assistente, os valores das horas e do valor por hora são atribuídos, e o método salario() é chamado, exibindo os respectivos pagamentos no console. Este código demonstra o uso de classes abstratas, herança e polimorfismo.

Construtor


// Definição da classe Produto
class Produto {
  // Atributos da classe
  String nome;
  double preco;
  int quantidade;

  // Construtor padrão
  Produto(this.nome, this.preco, this.quantidade);

  // Construtor nomeado para inicialização com preço padrão
  Produto.comPrecoPadrao(this.nome, this.quantidade)
    : preco = 10.0; // Define um preço padrão

  Produto.comNomePadrao(this.preco, this.quantidade) : nome = "Lápis";
  // Método para exibir informações do produto

  void exibirInformacoes() {
    print("Nome: $nome");
    print("Preço: R\$ ${preco.toStringAsFixed(2)}");
    print("Quantidade: $quantidade");
  }

  // Método para calcular o valor total do estoque do produto
  double calcularValorEstoque() {
    return preco * quantidade;
  }
}

void main() {
  // Criando objetos da classe Produto usando diferentes construtores
  Produto produto1 = Produto("Laptop", 2999.90, 5);
  Produto produto2 = Produto.comPrecoPadrao("Caderno", 20);
  Produto produto3 = Produto.comNomePadrao(1.50, 30);

  // Usando métodos dos objetos
  produto1.exibirInformacoes();
  print(
    "Valor do estoque: R\$ ${produto1.calcularValorEstoque().toStringAsFixed(2)}\n",
  );

  produto2.exibirInformacoes();
  print(
    "Valor do estoque: R\$ ${produto2.calcularValorEstoque().toStringAsFixed(2)}\n",
  );

  produto3.exibirInformacoes();
  print(
    "Valor do estoque: R\$ ${produto3.calcularValorEstoque().toStringAsFixed(2)}\n",
  );
}

Metodologia:

Este código utiliza a programação orientada a objetos (POO) para modelar um sistema de gerenciamento de produtos. A classe Produto encapsula atributos como nome, preco e quantidade, além de fornecer construtores para diferentes cenários de inicialização. Métodos como exibirInformacoes e calcularValorEstoque são responsáveis por exibir os detalhes do produto e calcular o valor total do estoque, respectivamente. A flexibilidade nos construtores permite criar objetos com valores padrão para atributos, promovendo reutilização e simplicidade. No método principal, vários objetos são instanciados, demonstrando a funcionalidade da classe e destacando a eficiência da abordagem.

class Pessoa {
  ///variáveis
  late double _peso;
  late double _altura;
  late double _imc;

  ///getter e setter
  double get peso => _peso;
  set peso(double nPeso) => _peso = nPeso;

  double get altura => _altura;
  set altura(double nAltura) => _altura = nAltura;

  double get imc => _imc;
  set imc(double nImc) => _imc - nImc;

  ///método calcular
  void calcular_imc() {
    this._imc = this._peso / (this._altura * this._altura);
  }

  ///método analisar
  void analisar_imc() {
    if (imc < 18.5) {
      print("Abaixo do peso");
    } else if (18.5 <= imc && imc < 24.9) {
      print("Peso normal");
    } else if (25 <= imc && imc < 29.9) {
      print("Sobrepeso");
    } else {
      print("Obesidade");
    }
  }
}

void main() {
  var imc = Pessoa();
  imc._altura = 1.75;
  imc._peso = 55;

  imc.calcular_imc();
  imc.analisar_imc();
  
   imc._altura = 1.40;
  imc._peso = 55;

  imc.calcular_imc();
  imc.analisar_imc();
  
   imc._altura = 1.45;
  imc._peso = 80;

  imc.calcular_imc();
  imc.analisar_imc();


}

Metodologia:

Este código utiliza a programação orientada a objetos (POO) para modelar um sistema de cálculo e análise do Índice de Massa Corporal (IMC) de uma pessoa. A classe Pessoa encapsula atributos como peso, altura e IMC, utilizando getters e setters para controlar o acesso e modificação dos valores. O método calcular_imc realiza o cálculo do IMC com base na fórmula padrão, enquanto o método analisar_imc interpreta o resultado e classifica o IMC em faixas como "Abaixo do peso", "Peso normal", "Sobrepeso" ou "Obesidade". O método principal (main) cria um objeto da classe Pessoa e simula diferentes cenários ao ajustar os valores de peso e altura, demonstrando a funcionalidade da classe e destacando sua eficiência na organização do código. Embora funcional, o código apresenta uma inconsistência no setter do atributo _imc e pode ser aprimorado com validações mais robustas.

class Imovel {
  //variáveis
  late double _largura;
  late double _comprimento;
  late double _area;

  //setter e getter
  double get altura => _largura;
  set altura(double nAltura) => _largura = nAltura;

  double get comprimento => _comprimento;
  set comprimento(double nComprimento) => _comprimento = nComprimento;

  double get area => _area;
  set area(double nArea) => _area = nArea;

  //método calculo área
  void calculo_area() {
    this._area = this._comprimento * this._largura;
  }

  void analise_iptu() {
    if (area <= 50 && area != 0) {
      print("500");
    } else if (51 <= area && area < 100) {
      print("800");
    } else if (100 <= area && area < 200) {
      print("1200");
    } else if (area > 200) {
      print("2000");
    } else {
      print("300");
    }
  }
}

void main() {
  var iptu = Imovel();
  iptu._largura = 0;
  iptu._comprimento = 0;

  iptu.calculo_area();
  iptu.analise_iptu();

  iptu._largura = 5;
  iptu._comprimento = 5;

  iptu.calculo_area();
  iptu.analise_iptu();

  iptu._largura = 3;
  iptu._comprimento = 30;

  iptu.calculo_area();
  iptu.analise_iptu();

  iptu._largura = 10;
  iptu._comprimento = 10;

  iptu.calculo_area();
  iptu.analise_iptu();

  iptu._largura = 30;
  iptu._comprimento = 30;

  iptu.calculo_area();
  iptu.analise_iptu();
}

Metodologia:

Este código utiliza a programação orientada a objetos (POO) para modelar um sistema de cálculo de área e imposto (IPTU) para imóveis. A classe Imovel encapsula atributos como largura, comprimento e área, utilizando getters e setters para controlar o acesso e modificação desses valores. Métodos como calculo_area e analise_iptu são responsáveis por calcular a área do imóvel e determinar o valor do IPTU com base em faixas pré-definidas. O método principal (main) cria um objeto da classe Imovel e simula diferentes cenários ao ajustar as dimensões do imóvel, demonstrando a funcionalidade da classe e destacando sua organização. Embora funcional, o código pode ser aprimorado com validações e nomes mais claros para maior legibilidade e robustez.
