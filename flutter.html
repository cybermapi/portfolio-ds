Função conversor(horas em minutos)


import 'package:flutter/material.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        appBar: AppBar(title: Text('Conversor de horas em minutos')),

        body: ConversorForm(),
      ),
    );
  }
}

class ConversorForm extends StatefulWidget {
  @override
  _ConversorFormState createState() => _ConversorFormState();
}

class _ConversorFormState extends State {
  final TextEditingController _numero1 = TextEditingController();

  String _resultado = '';

  void _converterMinutos() {
    final double? numero1 = double.tryParse(_numero1.text);

    if (numero1 != null) {
      final double conversaoMinutos = numero1 * 60;

      setState(() {
        _resultado = 'Resultado: $conversaoMinutos minutos';
      });
    } else {
      setState(() {
        _resultado = 'Por favor, insira números válidos.';
      });
    }
  }

  void _converterSegundos() {
    final double? numero1 = double.tryParse(_numero1.text);

    if (numero1 != null) {
      final double conversaoSegundos = numero1 * 3600;

      setState(() {
        _resultado = 'Resultado: $conversaoSegundos segundos';
      });
    } else {
      setState(() {
        _resultado = 'Por favor, insira números válidos.';
      });
    }
  }

  @override
  Widget build(BuildContext context) {
    return Padding(
      padding: const EdgeInsets.all(16.0),

      child: Column(
        mainAxisAlignment: MainAxisAlignment.center,

        children: [
          TextField(
            controller: _numero1,

            decoration: InputDecoration(labelText: 'Número 1'),

            keyboardType: TextInputType.numberWithOptions(decimal: true),
          ),

          SizedBox(height: 20),

          Row(
            mainAxisAlignment: MainAxisAlignment.center,

            children: [
              ElevatedButton(
                onPressed: _converterMinutos,
                child: Text('Minutos'),
              ),

              SizedBox(width: 20), // Espaçamento entre os botões

              ElevatedButton(
                onPressed: _converterSegundos,
                child: Text('Segundos'),
              ),

              SizedBox(height: 20),              
            ],
          ),
          SizedBox(height:20),
          
          Text(_resultado, style: TextStyle(fontSize: 24))
        ],
      ),
    );
  }
}

Calculadora de preço


import 'package:flutter/material.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        appBar: AppBar(title: Text('Calculadora de preço')),

        body: SomadorForm(),
      ),
    );
  }
}

class SomadorForm extends StatefulWidget {
  @override
  _SomadorFormState createState() => _SomadorFormState();
}

class _SomadorFormState extends State {
  final TextEditingController _valorProduto = TextEditingController();

  final TextEditingController _quantidade = TextEditingController();

  String _resultado = '';

  void _calcPreco() {
    final double? valor = double.tryParse(_valorProduto.text);

    final double? quantidade = double.tryParse(_quantidade.text);

    if (valor != null && quantidade != null) {
      final double preco = valor * quantidade;

      setState(() {
        _resultado = 'Resultado: R\$ $preco';
      });
    } else {
      setState(() {
        _resultado = 'Por favor, insira números válidos.';
      });
    }
  }

  @override
  Widget build(BuildContext context) {
    return Padding(
      padding: const EdgeInsets.all(16.0),

      child: Column(
        mainAxisAlignment: MainAxisAlignment.center,

        children: [
          TextField(
            controller: _valorProduto,

            decoration: InputDecoration(labelText: 'Valor do produto'),

            keyboardType: TextInputType.numberWithOptions(decimal: true),
          ),

          TextField(
            controller: _quantidade,

            decoration: InputDecoration(labelText: 'Quantidade do produto'),

            keyboardType: TextInputType.numberWithOptions(decimal: true),
          ),

          SizedBox(height: 20),

          Row(
            mainAxisAlignment: MainAxisAlignment.center,

            children: [
              ElevatedButton(onPressed: _calcPreco, child: Text('Calcular')),

              SizedBox(width: 20), // Espaçamento entre os botões
            ],
          ),

          SizedBox(height: 20),

          Text(_resultado, style: TextStyle(fontSize: 24)),
        ],
      ),
    );
  }
}

Calculadora de salário


import 'package:flutter/material.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        appBar: AppBar(title: Text('Calculadora de salário')),

        body: SomadorForm(),
      ),
    );
  }
}

class SomadorForm extends StatefulWidget {
  @override
  _SomadorFormState createState() => _SomadorFormState();
}

class _SomadorFormState extends State {
  final TextEditingController _valorHora = TextEditingController();

  final TextEditingController _qtdFaltas = TextEditingController();
  
  final TextEditingController _horasExtras = TextEditingController();
  
  String _resultado = '';

  void _salario() {
    final double? hora = double.tryParse(_valorHora.text);

    if (hora != null) {
      final double salario = hora * 8 * 30;

      setState(() {
        _resultado = 'Salário bruto: $salario';
      });
    } else {
      setState(() {
        _resultado = 'Por favor, insira números válidos.';
      });
    }
  }
  
  void _acrescimo() {
    final double? hora = double.tryParse(_valorHora.text);
    final double? extra = double.tryParse(_horasExtras.text);


    if (hora != null && extra != null) {
      final double acrescimo = hora * extra;

      setState(() {
        _resultado = 'Acréscimo salarial: $acrescimo';
      });
    } else {
      setState(() {
        _resultado = 'Por favor, insira números válidos.';
      });
    }
  }
  
  void _desconto() {
    final double? hora = double.tryParse(_valorHora.text);
    final double? faltas = double.tryParse(_qtdFaltas.text);

    if (hora != null && faltas != null) {
      final double descontos = hora * faltas * 8; //valor da hora x dias de falta x 8 horas de trabalho

      setState(() {
        _resultado = 'Descontos: $descontos';
      });
    } else {
      setState(() {
        _resultado = 'Por favor, insira números válidos.';
      });
    }
  }
  
  void _salarioFinal() {
    final double? hora = double.tryParse(_valorHora.text);
    final double? faltas = double.tryParse(_qtdFaltas.text);
    final double? extra = double.tryParse(_horasExtras.text);

    if (hora != null && faltas !=null && extra != null) {
      final double salario = (hora * 8 * 30) - (hora * faltas * 8) + (hora * extra);

      setState(() {
        _resultado = 'Salário líquido: $salario';
      });
    } else {
      setState(() {
        _resultado = 'Por favor, insira números válidos.';
      });
    }
  }

  @override
  Widget build(BuildContext context) {
    return Padding(
      padding: const EdgeInsets.all(16.0),

      child: Column(
        mainAxisAlignment: MainAxisAlignment.center,

        children: [
          TextField(
            controller: _valorHora,

            decoration: InputDecoration(labelText: 'Valor da hora'),

            keyboardType: TextInputType.numberWithOptions(decimal: true),
          ),

          TextField(
            controller: _horasExtras,

            decoration: InputDecoration(labelText: 'Horas extras'),

            keyboardType: TextInputType.numberWithOptions(decimal: true),
          ),
          
          TextField(
            controller: _qtdFaltas,

            decoration: InputDecoration(labelText: 'Quantidade de faltas'),

            keyboardType: TextInputType.numberWithOptions(decimal: true),
          ),

          SizedBox(height: 20),

          Row(
            mainAxisAlignment: MainAxisAlignment.center,

            children: [
              ElevatedButton(onPressed: _salario, child: Text('Salario bruto')),

              SizedBox(width: 20), // Espaçamento entre os botões
              
              ElevatedButton(onPressed: _acrescimo, child: Text('Acréscimo salarial')),
              
              SizedBox(width: 20),
              
              ElevatedButton(onPressed: _desconto, child: Text('Desconto salarial')),
              
              SizedBox(width: 20),
              
              ElevatedButton(onPressed: _salarioFinal, child: Text('Salário líquido')),
              
              SizedBox(width: 20),
              
              
            ],
          ),

          SizedBox(height: 20),

          Text(_resultado, style: TextStyle(fontSize: 24)),
        ],
      ),
    );
  }
}
